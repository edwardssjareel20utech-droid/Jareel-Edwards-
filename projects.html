1. Modular Arithmetic Calculator

Purpose: Demonstrates modular programming by using user-defined functions for basic math operations.


/*
Jareel Edwards 
*/
#include <stdio.h>

void calculate(int x, int y) {
    printf("Addition: %d\n", x + y);
    printf("Subtraction: %d\n", x - y);
    printf("Multiplication: %d\n", x * y);
}

int main() {
    int num1, num2;
    printf("Enter two numbers: ");
    scanf("%d %d", &num1, &num2);
    calculate(num1, num2);
    return 0;
}

2. Student ID Validation

Purpose: Validates that a UTech ID starts with '240' and is exactly 7 digits long.


/*
Jareel Edwards 
*/
#include <stdio.h>
#include <string.h>

int main() {
    char id[10];
    printf("Enter Student ID: ");
    scanf("%s", id);

    if (strlen(id) == 7 && strncmp(id, "240", 3) == 0) {
        printf("ID %s is Valid.\n", id);
    } else {
        printf("Invalid ID Format.\n");
    }
    return 0;
}

3. Array-Based Grade Average

Purpose: Stores 5 test scores in an array to calculate average and determine pass/fail status.


/*
Jareel Edwards 
*/
#include <stdio.h>

int main() {
    float grades[5], sum = 0, avg;
    for(int i = 0; i < 5; i++) {
        printf("Enter grade for Test %d: ", i + 1);
        scanf("%f", &grades[i]);
        sum += grades[i];
    }
    avg = sum / 5;
    printf("Average Score: %.2f\n", avg);
    printf("Status: %s\n", (avg >= 50) ? "PASS" : "FAIL");
    return 0;
}

4. Pointer-Based Memory Swapping

Purpose: Swaps two integers using pointers (pass-by-reference).


/*
Jareel Edwards 
*/                
#include <stdio.h>

void swap(int *n1, int *n2) {
    int temp = *n1;
    *n1 = *n2;
    *n2 = temp;
}

int main() {
    int x = 24, y = 42;
    printf("Before Swap: x=%d, y=%d\n", x, y);
    swap(&x, &y);
    printf("After Swap: x=%d, y=%d\n", x, y);
    return 0;
}

5. Structure-Based Record System

Purpose: Uses structs to manage courier data, relevant to logistics and database design.


/*
Jareel Edwards 
*/               
#include <stdio.h>

struct Courier {
    int packageID;
    float weight;
    char destination[20];
};

int main() {
    struct Courier c1 = {101, 5.5, "Kingston"};
    printf("Package ID: %d\n", c1.packageID);
    printf("Weight: %.1f kg\n", c1.weight);
    printf("Destination: %s\n", c1.destination);
    return 0;
}
